  def stats
    rows = query("select * from SPAMSTATS where phrase = ?", TOTAL_KEY)
    if (rows && rows.size > 0)
      spam = rows[0][1].to_i
      good = rows[0][2].to_i
      total = spam + good
    else
      good = spam = total = nil
    end
    rows = query("select count from SPAMSTATS")
    if (rows && rows.size > 0)
      phrases = rows[0][0].to_i
    else
      phrases = nil
    end
    {:phrases => phrases, :total => total, :spam => spam, :good => good }
  end
